-- Scripts de criação da tabela

CREATE TABLE Genero (
    GeneroId INT IDENTITY(1,1) PRIMARY KEY,
    Descricao VARCHAR(100) NOT NULL
);

CREATE TABLE Filme (
    FilmeId INT IDENTITY(1,1) PRIMARY KEY,
    Nome VARCHAR(200) NOT NULL,
    Duracao INT NOT NULL,
    GeneroId INT NOT NULL,
    FOREIGN KEY (GeneroId) REFERENCES Genero(GeneroId)
);

CREATE TABLE Ator (
    AtorId INT IDENTITY(1,1) PRIMARY KEY,
    Nome VARCHAR(150) NOT NULL
);

CREATE TABLE Elenco (
    ElencoId INT IDENTITY(1,1) NOT NULL,
    FilmeId INT NOT NULL,
    AtorId INT NOT NULL,
    PRIMARY KEY (FilmeId, AtorId),
    FOREIGN KEY (FilmeId) REFERENCES Filme(FilmeId),
    FOREIGN KEY (AtorId) REFERENCES Ator(AtorId)
);
